    private static double[] bernoulliPdf(int[] x, double p) {
        double[] probabilities = new double[x.length];

        for(int i=0; i<x.length; i++) {
            if(x[i]==0) {
                probabilities[i] = 1-p;
            } else {
                probabilities[i] = p;
            }
        }

        return probabilities;
    }
